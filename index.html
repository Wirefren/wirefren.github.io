#include <Wire.h>
#include <VL53L0X.h>

VL53L0X sensor;

void setup() {
  Serial.begin(9600);
  Wire.begin();
  
  // Initialize the sensor
  if (!sensor.init()) {
    Serial.println("Failed to initialize VL53L0X sensor!");
    while (1);
  }

  // Set the measurement range (Short, Medium, Long)
  // You can change this to VL53L0X::VL53L0X_RangingModeMedium or VL53L0X::VL53L0X_RangingModeLong
  sensor.setDistanceMode(VL53L0X::VL53L0X_RangingModeShort);

  // Start continuous measurements
  sensor.startContinuous();
}

void loop() {
  // Read the distance in millimeters
  uint16_t distance = sensor.readRangeContinuousMillimeters();

  // Check for measurement validity
  if (sensor.timeoutOccurred()) {
    Serial.println("Measurement timeout!");
  } else {
    Serial.print("Distance: ");
    Serial.print(distance);
    Serial.println(" mm");
  }

  delay(500);  // Delay between measurements
}
